551         var maxTowersPlaced = schoolBoardToTowersPlaced.values()
552                 .stream()
553                 .mapToInt(Integer::intValue)
554                 .max();
555 
556         //Every player that placed the maximum number of towers could be a winner
557         possibleWinningSchoolBoards = schoolBoardToTowersPlaced.entrySet()
558                 .stream()
559                 .filter(entry -> entry.getValue() == maxTowersPlaced.getAsInt())
560                 .map(Map.Entry::getKey)
561                 .toList();
562 
563 
564         if(maxTowersPlaced.isPresent() && maxTowersPlaced.getAsInt() >= this.strategy.getNumberOfTowers()){
565             //If there is a winner for the number of towers placed, then prepare the winners map with it.
566             for (var schoolBoard : this.schoolBoards)
567                 schoolBoardIdToIsWinnerMap.put(schoolBoard.getId(), possibleWinningSchoolBoards.contains(schoolBoard));
568 
569         } else if (this.onlyThreeArchipelagoLeft() || this.round.isLastRoundDone()) {
570             if (possibleWinningSchoolBoards.size() > 1) {
571                 if (this.numberOfPlayers < GameConstants.MAX_NUMBER_OF_PLAYERS.value) {
572                     // Filter by number of professors
573                     var maxNumberOfProfessors = possibleWinningSchoolBoards
574                             .stream()
575                             .mapToInt(schoolBoard -> schoolBoard.getProfessors().size())
576                             .max();
577 
578                     //possibleWinningSchoolBoards.removeIf(schoolBoard -> schoolBoard.getProfessors().size() < maxNumberOfProfessors.getAsInt());
579 
580                     possibleWinningSchoolBoards = possibleWinningSchoolBoards
581                             .stream()
582                             .filter(schoolBoard -> schoolBoard.getProfessors().size() == maxNumberOfProfessors.getAsInt())
583                             .toList();
584 
585                 } else {
586                     var whiteSchoolBoards = possibleWinningSchoolBoards
587                             .stream()
588                             .filter(schoolBoard -> schoolBoard.getTowerColor().equals(TowerColor.WHITE));
589 
590                     var blackSchoolBoards = possibleWinningSchoolBoards
591                             .stream()
592                             .filter(schoolBoard -> schoolBoard.getTowerColor().equals(TowerColor.BLACK));
593 
594                     var whiteProfessors = whiteSchoolBoards
595                             .mapToInt(schoolBoard -> schoolBoard.getProfessors().size())
596                             .sum();
597 
598                     var blackProfessors = blackSchoolBoards
599                             .mapToInt(schoolBoard -> schoolBoard.getProfessors().size())
600                             .sum();
601 
602                     possibleWinningSchoolBoards = (whiteProfessors > blackProfessors ? whiteSchoolBoards : blackSchoolBoards).toList();
603                 }
604             }
605             for (var schoolBoard : this.schoolBoards)
606                 schoolBoardIdToIsWinnerMap.put(schoolBoard.getId(), possibleWinningSchoolBoards.contains(schoolBoard));
607         }
608         return schoolBoardIdToIsWinnerMap;
609     }

949     public Optional<Integer> getMostInfluentSchoolBoardId(List<Integer> archipelagoIslandCodes){

954             orderedPlayersInfluences = influenceMap.get().entrySet()
955                     .stream()
956                     .sorted(Map.Entry.comparingByValue(Comparator.reverseOrder())) //sorts the map using the influences in descending order

960             return Optional.empty();

963

968                 return Optional.of(orderedPlayersInfluences.get(0).getKey());

971         }
972         //if the number of players is 4, the most influent is calculated between the most influent and the third most influent, since the second most influent is certainly a teammate.
